# 6. Event Handling

## Concept

- React uses camelCase for event handlers (e.g., `onClick`).
- Event handlers are functions that respond to user actions.

## Why it matters in interviews

- Handling events is essential for interactive UIs (buttons, forms, etc.).

## Example

```jsx
function ClickButton() {
  function handleClick() {
    alert('Button clicked!');
  }

  return <button onClick={handleClick}>Click Me</button>;
}
```

## Key Points

- You can pass arguments to event handlers using arrow functions:  
  `<button onClick={() => handleClick(id)} />`
- React events are synthetic (cross-browser compatible).

## Interview Questions & Answers

### Q1: What's the difference between synthetic events and native browser events?

**Answer:** React wraps native browser events in a "SyntheticEvent" system:
- **Synthetic Events:** Cross-browser compatible wrapper around native events
- **Native Events:** Direct browser events
- React attaches all events to the root element (event delegation)

```jsx
function EventComparison() {
  const handleSyntheticClick = (e) => {
    console.log('Synthetic event:', e);
    console.log('Type:', e.type); // "click"
    console.log('Pooled:', e._disposed); // Events are pooled in React 16
  };

  const handleNativeClick = () => {
    document.addEventListener('click', (nativeEvent) => {
      console.log('Native event:', nativeEvent);
      // Direct browser event object
    });
  };

  return (
    <div>
      <button onClick={handleSyntheticClick}>Synthetic Event</button>
      <button onClick={handleNativeClick}>Native Event</button>
    </div>
  );
}
```

**Key Difference:** 
- Synthetic events are **pooled** (reused) in React < 17, nullified after callback
- React 17+ doesn't pool events anymore

### Q2: How do you pass arguments to event handlers?

**Answer:** Use arrow functions or `.bind()`

```jsx
function EventWithArguments() {
  const [message, setMessage] = useState('');

  // Method 1: Arrow function (Most Common) ✅
  const handleClick1 = (id, name) => {
    console.log(`Clicked: ${id}, ${name}`);
    setMessage(`ID: ${id}, Name: ${name}`);
  };

  // Method 2: Using .bind()
  const handleClick2 = function(id) {
    console.log(this); // 'this' context is bound
  };

  // Method 3: Inline arrow function (Less Efficient)
  const handleClick3 = (id) => setMessage(`Item ${id} selected`);

  return (
    <div>
      {/* ✅ BEST: Pass arguments clearly */}
      <button onClick={() => handleClick1(1, 'Product A')}>
        Method 1: Arrow Function
      </button>

      {/* ⚠️ OK: But creates new function every render */}
      <button onClick={() => handleClick2.call(null, 2)}>
        Method 2: Bind
      </button>

      {/* ⚠️ AVOID: Inline logic */}
      <button onClick={() => handleClick3(3)}>
        Method 3: Inline Arrow
      </button>

      <p>{message}</p>
    </div>
  );
}
```

**Interview Note:** Arrow functions create new function instances on every render, but for event handlers it's usually fine. Use `useCallback` for optimization if needed.

### Q3: What's the difference between `onClick={handleClick}` and `onClick={() => handleClick()}`?

**Answer:** 

```jsx
function ClickDifference() {
  const handleClick = () => {
    alert('Clicked!');
  };

  return (
    <div>
      {/* ✅ GOOD: Function reference - no new function created */}
      <button onClick={handleClick}>Method 1: Reference</button>

      {/* ⚠️ OK: Arrow function - new function created each render */}
      <button onClick={() => handleClick()}>Method 2: Arrow Function</button>

      {/* ❌ WRONG: Executes immediately on render */}
      <button onClick={handleClick()}>Method 3: Direct Call</button>
    </div>
  );
}

// For performance:
// Method 1 > Method 2 > Method 3
```

### Q4: Interview Scenario - Form Submission & Event Handling

```jsx
function LoginForm() {
  const [formData, setFormData] = useState({
    email: '',
    password: '',
    rememberMe: false
  });
  const [errors, setErrors] = useState({});
  const [isSubmitting, setIsSubmitting] = useState(false);

  // Handle input change
  const handleInputChange = (e) => {
    const { name, value, type, checked } = e.target;
    setFormData(prev => ({
      ...prev,
      [name]: type === 'checkbox' ? checked : value
    }));
    // Clear error for this field
    if (errors[name]) {
      setErrors(prev => ({ ...prev, [name]: '' }));
    }
  };

  // Validate form
  const validateForm = () => {
    const newErrors = {};
    if (!formData.email) newErrors.email = 'Email required';
    if (!formData.password) newErrors.password = 'Password required';
    if (formData.email && !formData.email.includes('@')) {
      newErrors.email = 'Invalid email';
    }
    return newErrors;
  };

  // Handle form submission
  const handleSubmit = async (e) => {
    e.preventDefault(); // ← Important! Prevents page reload
    
    const validationErrors = validateForm();
    if (Object.keys(validationErrors).length > 0) {
      setErrors(validationErrors);
      return;
    }

    setIsSubmitting(true);
    try {
      // Simulate API call
      await new Promise(resolve => setTimeout(resolve, 1000));
      alert('Login successful!');
      setFormData({ email: '', password: '', rememberMe: false });
    } catch (error) {
      setErrors({ submit: 'Login failed' });
    } finally {
      setIsSubmitting(false);
    }
  };

  return (
    <form onSubmit={handleSubmit}>
      <div>
        <input
          type="email"
          name="email"
          placeholder="Email"
          value={formData.email}
          onChange={handleInputChange}
        />
        {errors.email && <p style={{ color: 'red' }}>{errors.email}</p>}
      </div>

      <div>
        <input
          type="password"
          name="password"
          placeholder="Password"
          value={formData.password}
          onChange={handleInputChange}
        />
        {errors.password && <p style={{ color: 'red' }}>{errors.password}</p>}
      </div>

      <label>
        <input
          type="checkbox"
          name="rememberMe"
          checked={formData.rememberMe}
          onChange={handleInputChange}
        />
        Remember me
      </label>

      <button type="submit" disabled={isSubmitting}>
        {isSubmitting ? 'Logging in...' : 'Login'}
      </button>

      {errors.submit && <p style={{ color: 'red' }}>{errors.submit}</p>}
    </form>
  );
}
```

### Q5: Event Delegation & Multiple Items

```jsx
function TodoListWithEvents() {
  const [todos, setTodos] = useState([
    { id: 1, text: 'Learn React', completed: false },
    { id: 2, text: 'Build Projects', completed: false },
    { id: 3, text: 'Get Job', completed: false }
  ]);

  const [newTodo, setNewTodo] = useState('');

  // Handle todo completion
  const handleToggleTodo = (id) => {
    setTodos(todos.map(todo =>
      todo.id === id ? { ...todo, completed: !todo.completed } : todo
    ));
  };

  // Handle todo deletion
  const handleDeleteTodo = (id) => {
    setTodos(todos.filter(todo => todo.id !== id));
  };

  // Handle adding new todo
  const handleAddTodo = (e) => {
    e.preventDefault();
    if (newTodo.trim()) {
      setTodos([
        ...todos,
        { id: Date.now(), text: newTodo, completed: false }
      ]);
      setNewTodo('');
    }
  };

  // Handle input change
  const handleInputChange = (e) => {
    setNewTodo(e.target.value);
  };

  // Handle keyboard (Enter key)
  const handleKeyPress = (e) => {
    if (e.key === 'Enter') {
      handleAddTodo(e);
    }
  };

  return (
    <div>
      <form onSubmit={handleAddTodo}>
        <input
          type="text"
          value={newTodo}
          onChange={handleInputChange}
          onKeyPress={handleKeyPress}
          placeholder="Add a new todo..."
        />
        <button type="submit">Add</button>
      </form>

      <ul>
        {todos.map(todo => (
          <li key={todo.id}>
            <input
              type="checkbox"
              checked={todo.completed}
              onChange={() => handleToggleTodo(todo.id)}
            />
            <span style={{
              textDecoration: todo.completed ? 'line-through' : 'none'
            }}>
              {todo.text}
            </span>
            <button onClick={() => handleDeleteTodo(todo.id)}>
              Delete
            </button>
          </li>
        ))}
      </ul>
    </div>
  );
}
```

### Q6: Event Bubbling & Event Capturing Interview Question

```jsx
function EventBubbling() {
  const handleParentClick = (e) => {
    console.log('Parent clicked');
  };

  const handleChildClick = (e) => {
    console.log('Child clicked');
    // e.stopPropagation(); // Prevents bubbling up to parent
  };

  const handleGrandchildClick = (e) => {
    console.log('Grandchild clicked');
    e.stopPropagation(); // Stops event from reaching parent
  };

  return (
    <div onClick={handleParentClick} style={{ padding: '20px', border: '2px solid red' }}>
      Parent
      <div onClick={handleChildClick} style={{ padding: '20px', border: '2px solid blue' }}>
        Child
        <button onClick={handleGrandchildClick}>
          Grandchild (stopPropagation)
        </button>
      </div>
    </div>
  );
}

// Click order WITHOUT stopPropagation:
// Grandchild → Child → Parent
//
// Click order WITH stopPropagation on grandchild:
// Grandchild → (stops here, doesn't reach Child or Parent)
```

**Interview Question:** "What happens when you click the grandchild button?"
- **With `e.stopPropagation()`:** Only "Grandchild clicked" logs
- **Without it:** "Grandchild clicked", "Child clicked", "Parent clicked" all log

### Q7: Preventing Default Behavior

```jsx
function PreventDefault() {
  const handleLinkClick = (e) => {
    e.preventDefault(); // Prevents navigation
    alert('Navigation prevented!');
  };

  const handleFormSubmit = (e) => {
    e.preventDefault(); // Prevents page reload
    console.log('Form submitted without reload');
  };

  const handleContextMenu = (e) => {
    e.preventDefault(); // Prevents right-click menu
    alert('Right-click prevented!');
  };

  return (
    <div>
      <a href="https://google.com" onClick={handleLinkClick}>
        Click me (won't navigate)
      </a>

      <form onSubmit={handleFormSubmit}>
        <button type="submit">Submit (won't reload page)</button>
      </form>

      <div onContextMenu={handleContextMenu}>
        Right-click here (menu prevented)
      </div>
    </div>
  );
}
```

function MouseEventDemo() {
  const [mousePos, setMousePos] = useState({ x: 0, y: 0 });
  const [isHovering, setIsHovering] = useState(false);

  const handleMouseMove = (e) => {
    setMousePos({ x: e.clientX, y: e.clientY });

    setIsHovering(true);
  };

  const handleMouseLeave = () => {
    setIsHovering(false);
  };

    alert('Double clicked!');
  };

  return (
    <div
      onMouseMove={handleMouseMove}
      onMouseEnter={handleMouseEnter}
      onMouseLeave={handleMouseLeave}
      onDoubleClick={handleDoubleClick}
      style={{
        padding: '20px',
        border: '2px solid black',
        backgroundColor: isHovering ? '#f0f0f0' : '#fff'
      }}
    >
      <p>Mouse Position: X={mousePos.x}, Y={mousePos.y}</p>
      <p>{isHovering ? 'Hovering' : 'Not hovering'}</p>
    </div>
  );
}

### Q9: Performance Optimization - useCallback

```jsx
function OptimizedEventHandling() {
  const [count, setCount] = useState(0);

  // ❌ PROBLEM: New function created on every render
  // Child components re-render even if handler reference changes
  const handleClickBad = () => {
    setCount(count + 1);
  };

  // ✅ SOLUTION: useCallback memoizes the function
  const handleClickGood = useCallback(() => {
    setCount(prev => prev + 1);
  }, []);

  return (
    <div>
      <button onClick={handleClickGood}>Count: {count}</button>
      <ChildComponent onCustomClick={handleClickGood} />
    </div>
  );
}

function ChildComponent({ onCustomClick }) {
  console.log('ChildComponent rendered');
  return <button onClick={onCustomClick}>Child Button</button>;
}
```

## Interview Mistakes to Avoid

```jsx
// ❌ MISTAKE 1: Calling function directly (executes on render!)
<button onClick={handleClick()}>Click</button>

// ✅ CORRECT: Pass function reference
<button onClick={handleClick}>Click</button>

// ❌ MISTAKE 2: Not preventing default
<form onSubmit={handleSubmit}>  {/* Page will reload! */}
  <button type="submit">Submit</button>
</form>

// ✅ CORRECT: Use preventDefault
const handleSubmit = (e) => {
  e.preventDefault();
  // Your logic
};

// ❌ MISTAKE 3: Forgetting to pass event object
const handleChange = () => {
  // e.target is undefined!
};
<input onChange={handleChange} />

// ✅ CORRECT: Receive event object
const handleChange = (e) => {
  console.log(e.target.value);
};
<input onChange={handleChange} />

// ❌ MISTAKE 4: Not using useCallback for child prop
const MyButton = ({ onClick }) => {
  // Re-renders every time parent renders (inefficient)
  return <button onClick={onClick}>Click</button>;
};

// ✅ CORRECT: Memoize handler
const handleClick = useCallback(() => {
  // handler logic
}, []);
```

## Real-World Interview Scenario: Search & Filter

```jsx
function SearchableList() {
  const [items] = useState([
    { id: 1, name: 'JavaScript', category: 'Language' },
    { id: 2, name: 'React', category: 'Library' },
    { id: 3, name: 'Node.js', category: 'Runtime' },
    { id: 4, name: 'TypeScript', category: 'Language' }
  ]);

  const [searchTerm, setSearchTerm] = useState('');
  const [selectedCategory, setSelectedCategory] = useState('');
  const [selectedItems, setSelectedItems] = useState(new Set());

  const handleSearch = (e) => {
    setSearchTerm(e.target.value.toLowerCase());
  };

  const handleCategoryChange = (e) => {
    setSelectedCategory(e.target.value);
  };

  const handleSelectItem = (id) => {
    const newSelected = new Set(selectedItems);
    if (newSelected.has(id)) {
      newSelected.delete(id);
    } else {
      newSelected.add(id);
    }
    setSelectedItems(newSelected);
  };

  const handleSelectAll = (e) => {
    if (e.target.checked) {
      const allIds = new Set(filteredItems.map(item => item.id));
      setSelectedItems(allIds);
    } else {
      setSelectedItems(new Set());
    }
  };

  const filteredItems = items.filter(item => {
    const matchesSearch = item.name.toLowerCase().includes(searchTerm);
    const matchesCategory = !selectedCategory || item.category === selectedCategory;
    return matchesSearch && matchesCategory;
  });

  return (
    <div>
      <input
        type="text"
        placeholder="Search..."
        onChange={handleSearch}
        value={searchTerm}
      />

      <select value={selectedCategory} onChange={handleCategoryChange}>
        <option value="">All Categories</option>
        <option value="Language">Language</option>
        <option value="Library">Library</option>
        <option value="Runtime">Runtime</option>
      </select>

      <label>
        <input
          type="checkbox"
          onChange={handleSelectAll}
          checked={selectedItems.size === filteredItems.length && filteredItems.length > 0}
        />
        Select All
      </label>

      <ul>
        {filteredItems.map(item => (
          <li key={item.id}>
            <input
              type="checkbox"
              checked={selectedItems.has(item.id)}
              onChange={() => handleSelectItem(item.id)}
            />
            <span>{item.name} - {item.category}</span>
          </li>
        ))}
      </ul>

      <p>Selected: {selectedItems.size}</p>
    </div>
  );
}
```
